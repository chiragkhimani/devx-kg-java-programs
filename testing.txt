window.__toastLog = [];
new MutationObserver((mutations) => {
  mutations.forEach(m => {
    if (m.type === 'childList') {
      m.addedNodes.forEach(n => {
        if (n.nodeType === 1) window.__toastLog.push(n.outerHTML);
      });
    }
    if (m.type === 'attributes') {
      window.__toastLog.push(m.target.outerHTML);
    }
  });
}).observe(document.body, {
  childList: true,
  subtree: true,
  attributes: true,
  attributeFilter: ['class', 'style']
});
